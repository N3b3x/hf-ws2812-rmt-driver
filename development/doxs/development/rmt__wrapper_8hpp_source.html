165 esp_err_t transmit_bytes(const uint8_t* data, size_t length, const rmt_symbol_word_t& bit0,
166 const rmt_symbol_word_t& bit1, TickType_t timeout = portMAX_DELAY) {
167 // Create a throw‑away bytes encoder (kept only for this call)
168 rmt_bytes_encoder_config_t be_cfg = {};
169 be_cfg.bit0 = bit0;
170 be_cfg.bit1 = bit1;
171 be_cfg.flags.msb_first = true;

173 ESP_RETURN_ON_ERROR(rmt_new_bytes_encoder(&be_cfg, &bytes_enc), "RmtTx", "new bytes enc");
174
175 rmt_transmit_config_t tx_cfg = {};
176 tx_cfg.loop_count = 0;
177 esp_err_t err = rmt_transmit(_handle, bytes_enc, data, length, &tx_cfg);
178 if (err == ESP_OK)
179 err = rmt_tx_wait_all_done(_handle, timeout);
180 rmt_del_encoder(bytes_enc);
181 return err;
182 }

189 esp_err_t transmit_ws2812(const uint8_t* grb, size_t length, TickType_t timeout = portMAX_DELAY) {
190 if (!_ws_encoder) {
191 rmt_bytes_encoder_config_t ws_cfg = {};
192 ws_cfg.bit0 = make_ws2812_bit0();
193 ws_cfg.bit1 = make_ws2812_bit1();
194 ws_cfg.flags.msb_first = true;
195 ESP_RETURN_ON_ERROR(rmt_new_bytes_encoder(&ws_cfg, &_ws_encoder), "RmtTx", "new ws2812 enc");
196 }
197 rmt_transmit_config_t tx_cfg = {};
198 tx_cfg.loop_count = 0;
199 esp_err_t err = rmt_transmit(_handle, _ws_encoder, grb, length, &tx_cfg);
200 if (err == ESP_OK)
201 err = rmt_tx_wait_all_done(_handle, timeout);
202 return err;
203 }

210 static constexpr rmt_symbol_word_t make_ws2812_bit0(uint32_t resolution_hz = 10'000'000) {
211 // T0H=0.4 μs, T0L≈0.85 μs
212 uint32_t ticks_h = (400'000'000ULL / resolution_hz + 9) / 10; // round‑nearest
213 uint32_t ticks_l = (850'000'000ULL / resolution_hz + 9) / 10;
214 return {
215 .duration0 = (uint16_t)ticks_h, .level0 = 1, .duration1 = (uint16_t)ticks_l, .level1 = 0};
216 }
217
218 static constexpr rmt_symbol_word_t make_ws2812_bit1(uint32_t resolution_hz = 10'000'000) {
219 // T1H=0.8 μs, T1L≈0.45 μs
220 uint32_t ticks_h = (800'000'000ULL / resolution_hz + 9) / 10;
221 uint32_t ticks_l = (450'000'000ULL / resolution_hz + 9) / 10;
222 return {
223 .duration0 = (uint16_t)ticks_h, .level0 = 1, .duration1 = (uint16_t)ticks_l, .level1 = 0};
224 }
225
226 rmt_channel_handle_t _handle = nullptr;
227 rmt_encoder_handle_t _copy_encoder = nullptr;
228 rmt_encoder_handle_t _ws_encoder = nullptr;
229};

255 ESP_ERROR_CHECK(rmt_rx_register_event_callbacks(_handle, &cbs, this));
256 ESP_ERROR_CHECK(rmt_enable(_handle));

259 _queue = xQueueCreate(4, sizeof(size_t));
260
261 // Configure RX receive behavior (can be changed later)
262 _rcv_cfg.signal_range_max_ns = idle_threshold_us * 1000ULL;
263 _rcv_cfg.signal_range_min_ns = filter_ns;
264 }

290 esp_err_t receive(rmt_symbol_word_t* buffer, size_t buffer_symbols, size_t* out_symbols,
291 TickType_t timeout = portMAX_DELAY) {
292 if (!buffer || !out_symbols)
293 return ESP_ERR_INVALID_ARG;
294
295 esp_err_t err =
296 rmt_receive(_handle, buffer, buffer_symbols * sizeof(rmt_symbol_word_t), &_rcv_cfg);
297 if (err != ESP_OK)
298 return err;
299
300 // Wait until ISR posts the symbol count
301 if (xQueueReceive(_queue, out_symbols, timeout) != pdPASS) {
302 return ESP_ERR_TIMEOUT;
303 }
304 return ESP_OK;
305 }

309 }

312 static bool rx_done_cb_static(rmt_channel_handle_t /*chan*/,
313 const rmt_rx_done_event_data_t* edata, void* user_ctx) {
314 auto* self = static_cast<RmtRx*>(user_ctx);
315 BaseType_t HPW = pdFALSE;
316 xQueueSendFromISR(self->_queue, &edata->num_symbols, &HPW);
317 return HPW == pdTRUE; // context switch request
318 }
